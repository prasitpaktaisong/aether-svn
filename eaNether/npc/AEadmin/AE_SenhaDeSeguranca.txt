prontera,152,193,3	script	Íris, a Guardiã	745,{
set .@npcName$,"^FF0000[Guardiã da Segurança]^000000";

mes .@npcName$;
mes "Bem-Vindo! Eu sou a ^FF0000Guardiã de Segurança^000000.";
mes "Como tal, é meu trabalho garantir a segurança das Contas.";
mes "Mas é claro, se eles desejarem usar nosso serviço.";
next;
mes "Então, o que você gostaria de fazer?";
goto LacctCheck;
  LacctCheck:
  menu "• Habilitar/Desabilitar Segurança",iAD,"• Mudar Senha",iCP,"• Mudar frase de recuperação",iRP,"• Ver tentativas de senha",iVP,"• Cancelar",cancel;

iAD:
if (#SecurityPass$!="Sem PopUp"){
	set #SecurityPass$,"Sem PopUp";
	mes "Segurança agora está: ^FF0000Desligada^000000.";
	close;
}

if (#SecurityPass$=="Sem PopUp"){
	If (@AE_Seguro == 1){
		mes "Segurança agora está: ^0000FFLigada^000000.";
		set #SecurityPass$,@CSpassword$;
		close;
	}
	If (@AE_Seguro != 1){
		next;
		mes "Coloque sua nova Senha"; 
		input .@CSpassword$;
		if (.@CSpassword$==#RecoveryPhrase$){
			next;
			mes "Sua senha não pode ser igual à sua frase de recuperação! Por favor, tente novamente!";
			goto iAD;
		}
		set @CSpassword$,.@CSpassword$;
		next;
		
		mes "Digite sua senha novamente para confirmação.";
		input .@CSpassword$;
		if (.@CSpassword$!=@CSpassword$){
			mes "Está incorreto. Tente novamente.";
			close;
		}
		set #SecurityPass$,@CSpassword$;
		mes "Sistema de Segurança agora está ligado.";
		next;
		goto iRecoveryPhrase;
	
	iRecoveryPhrase:
		mes "Agora, por favor, digite uma frase de Recuperação de Senha";
		mes "Essa frase será usada para Recuperar sua Senha, no caso de você esquecê-la.";
		input .@RecoveryPhrase$;
		
		if (.@RecoveryPhrase$==#SecurityPass$){
			next; mes "Sua frase de recuperação não pode ser igual sua senha. Por favor, tente novamente.";
			next; goto iRecoveryPhrase;
		}
		set @RecoveryPhrase$,.@RecoveryPhrase$;
		next;
		
		mes "Coloque sua frase de recuperação denovo para confirmação.";
		input .@RecoveryPhrase$;
		
		if (@RecoveryPhrase$!=.@RecoveryPhrase$){
			mes "As frases não combinam. Por favor, tente novamente.";
			close;
		}
		
		set #RecoveryPhrase$,@RecoveryPhrase$;
		mes "Por favor, escreva a frase abaixo. Ela será usada para recuperar sua senha caso você a esqueça.";
		next;
		
		mes "Mais uma vez, por favor:";
		mes "^0000FF"+#RecoveryPhrase$+"^000000";
		mes "Escreva abaixo a Frase de Recuperação de Senha! ";
		goto iSecurityPhrase;
		
		iSecurityPhrase:
		next;
		mes "Agora coloque uma Frase de Segurança. Isso é um pouco diferente da sua frase de recuperação.";
		mes "Por favor, anote-a. Ela será usada para notificar o GM que você esqueceu sua Senha e Frase de Recuperação.";
		input .@SecurityPhrase$;
		if (.@SecurityPhrase$==#RecoveryPhrase$ || .@SecurityPhrase$==#SecurityPass$){next; mes "Sua Frase de Segurança não pode ser a mesma que sua Senha ou Frase de Recuperação"; goto iSecurityPhrase;}
		set #SecurityPhrase,.@SecurityPhrase;
		mes "Tenha certeza de manter essa frase segura, e não dividí-la com outras pessoas!";
		set @AE_Seguro,1;
		close;
	}
	end;
}

iCP:
	next;
	mes .@npcName$;
	if (#SecurityPass$=="Sem PopUp"){
		mes "Você não tem uma Senha definida.";
		mes "Por favor, defina uma ativando a segurança!";
		close;
	}
	mes "Muito bem. Por favor coloque sua Senha de Segurança atual.";
	input .@CSpassword$;
	if (.@CSpassword$!=#SecurityPass$){
		next;
		mes "Está incorreto. Tente novamente.";
		close;
	}
	next;
	mes "Agora, por favor digite sua nova Senha de Segurança.";
	input .@NSpassword$;
	if (.@NSpassword$==#RecoveryPhrase$){
		next;
		mes "Sua Senha não pode ser a mesma que sua Frase de Recuperação. Por favor, tente novamente.";
		goto iCP;
	}
	set @NSpassword$,.@NSpassword$;
	next;
	mes "Por favor, digite-a denovo para confirmação.";
	input .@CNSpassword$;
	if (@NSpassword$!=.@CNSpassword$){next; mes "Está incorreto. Tente novamente."; close;}
	set #SecurityPass$,.@CNSpassword$;
		next;
	mes "Sua Senha foi alterada com sucesso!";
	close;

iRP:
	next;
	mes .@npcName$;
	if (#RecoveryPhrase$==""){
		mes "Você não tem uma frase de recuperação definida.";
		mes "Por favor, defina uma ativando a segurança.";
		close;
	}		

	mes "Muito bem. Por favor digite sua Frase de Recuperação atual.";
	input .@RecovPhrase$;
	if (.@RecovPhrase$!=#RecoveryPhrase$){
		next;
		mes "Está incorreto! Tente novamente.";
		close;
	}
	next;

	mes "Agora, por favor digite sua nova Frase de Recuperação.";
	input .@NRP$;
	if (.@NRP$==#SecurityPass$){
		next; mes "Sua Frase de Recuperação não pode ser a mesma que sua Senha! Por favor, tente novamente.";
		goto iRP;
	}
	set @NRP$,.@NRP$;
	next;

	mes "Por favor, digite novamente para confirmação.";
	input .@NRP$;
	if (@NRP$!=.@NRP$){next; mes "As Frases não combinam. Tente novamente."; close;}
		set #RecoveryPhrase$,@NRP$;
	next;

	mes "Sua Frase de Recuperação foi alterada com sucesso.";
	mes "Por favor, anote-a. Ela será usada caso você esqueça sua senha.";
	next;

	mes "Aqui está:";
	mes "^0000FF"+#RecoveryPhrase$+"^000000";
	mes "Anote !";
	close;

iVP:
	next;
	mes .@npcName$;
	mes "Houveram no total: ^0000FF"+#SecurityAttempt+"^000000 tentativas de logar.";
	close;
cancel:
	close;

OnPCLoginEvent:
	set .@npcName$,"^FF0000[Guardiã da Segurança]^000000";
	set @BanTime,5;
	if (#SecurityPass$==""){
		mes .@npcName$;
		mes "Esse servidor tem um Sistema único, que permite a você ter uma segunda Senha.";
		mes "É claro, isso é opcional. Você gostaria de usá-la?";
		menu "• Sim, defina-a.",iYes,"• Não, e pare de mostrar isso, por favor.",iNo;
		iYes:
		next;
		callfunc "Set_AcctSecure";
		iNo:
		set #SecurityPass$,"Sem PopUp";
		//percentheal 100,100;
		mes "Bem. Não esqueça, se você estiver interessado, fale com a ^FF0000Guardiã da Segurança^000000, em ^0000FFProntera^000000.";
		close;
		end;
	}
	
if (#SecurityPass$=="Sem PopUp"){set @SecurityPass,1; end;}
if (#SecurityPass$!="Sem PopUp" && #SecurityPass$==#SecurityPass$){
	atcommand "@option 0 0 4 "+strcharinfo(0)+"";
	atcommand "@mute 5000 "+strcharinfo(0)+"";
	//sc_start 112,999999999,100;
	//sc_start 1,999999999,100;
	//sc_start 8,999999999,100;
	goto InputSecurityPass;
	}
end;

OnPCLoadMapEvent:
getmapxy(@Map$,@X,@Y,0);
if (@Map$=="Prontera" || @Map$!="Prontera"){
	if (@SecurityPass==0 && #SecurityPass$!="Sem PopUp" && #SecurityPass$==#SecurityPass$){
		goto InputSecurityPass;
	}
	if (@SecurityPass==0 && #SecurityPass$=="Sem PopUp"){
		set @SecurityPass,1;
		end;
	}
	if (@SecurityPass==1){
		end;
	}
	end;
}	 

InputSecurityPass:
	mes "Essa conta está protegida pela ^FF0000Guardiã da Segurança^000000.";
	mes "Coloque sua Senha para continuar jogando.";
	mes "Ou";
	mes "Coloque sua Frase de Recuperação, caso você esqueça sua senha.";
	mes "Ou";
	mes "Coloque sua Frase de Segurança caso você esqueça as duas acima.";
	input .@spass$;

if (#RecoveryPhrase$==.@spass$){
	next;
	mes "Aqui está sua senha, não perca-a denovo.";
    mes "^0000FF"+#SecurityPass$+"^000000";
    next;
	goto InputSecurityPass;
	}

	if (#SecurityPass$==.@spass$){
	mes "Está correto"; set @SecurityPass,1;
	set .@conta,getcharid(3);
	atcommand "@unmute "+rid2name(.@conta);
	attachrid(.@conta);
	atcommand "@option 0 0 0 "+strcharinfo(0)+"";
	//sc_end 1;
	//sc_end 112;
	//sc_end 8;
	//percentheal 100,100;
	close;
	end;
	}
	if (#SecurityPhrase$==.@spass$){
		for(set .@i,0; .@i<getarraysize($LP_ACCT_ID);
			set .@i,.@i+1){
				if (!$LP_ACCT_ID[.@i]){
				setarray $LP_ACCT_ID[.@i],getcharid(3,strcharinfo(0));
				setarray $LP_USER_NM[.@i],strcharinfo(0);
				next;
			}
		}	
		mes "Seu nome foi registrado na lista de Senhas perdidas.";
		mes "Suas informações serão reestabelecidas.";
		atcommand "@kick "+strcharinfo(0)+""; close;
	}
	   
if (#SecurityPass$!=.@spass$ && #RecoveryPhrase$!=.@spass$ && #SecurityPhrase$!=.@spass$){
	set @Attempt,@Attempt+1; mes "INCORRETO!!"; 
    if (@Attempt>=3){
		atcommand "@ban "+@BanTime+"mn "+strcharinfo(0)+"";
		end;
	} 
	next;
	goto InputSecurityPass;
	}
		
close;
end;
}

function	script	Set_AcctSecure	{
set .@npcName$,"^FF0000[Guardiã da Segurança]^000000";
menu "• Defina uma Senha de segurança",iSP,"• Informação no Sistema",iSI,"Cancelar",cancel;
  iSP:
	next;
	mes .@npcName$;
	mes "Muito bem. Por favor, digite uma Senha.";
	mes "Ela pode conter de 4 a 32 caracteres alfanuméricos";
	mes "(A-Z) & (1-0) são caracteres alfanuméricos.";
	input .@Npassword$;
	set @Npassword$,.@Npassword$;
	next;
	mes "Por favor digite sua Senha novamente para confirmação.";
	input .@Npassword$;
	if (.@Npassword$!=@Npassword$){
		mes "As Senhas não combinam. Tente novamente falando com a ^FF0000Guardiã da Segurança^000000.";
		goto iSP;
	}
	set #SecurityPass$,@Npassword$;
	next;
	mes "Sua Senha foi definida com sucesso.";
	next;
	goto iRecoveryPhrase;
	iRecoveryPhrase:
	mes "Agora, defina uma Frase de Recuperação, por favor.";
	mes "Essa frase será usada caso você esqueça sua Senha.";
	input .@RecovPhrase$;
	if (.@RecovPhrase$==#SecurityPass$){
		mes "Sua Frase de Recuperação não pode ser a mesma que sua Senha. Por favor, tente novamente.";
		next;
		goto iRecoveryPhrase;
	}
	set @RecovPhrase$,.@RecovPhrase$;
	next;
	mes "Por favor, digite sua Frase de Recuperação novamente para confirmação.";
	input .@RecovPhrase$;
	if (@RecovPhrase$!=.@RecovPhrase$){
		mes "As Frases de Recuperação não combinam. Tente novamente falando com a ^FF0000Guardiã da Segurança^000000.";
		next;
		goto iRecoveryPhrase;
	}
	
	set #RecoveryPhrase$,@RecovPhrase$;
	next;
	mes "Sua Frase de Recuperação foi definida com sucesso.";
	mes "Por favor, anote-a. Ela será usada se você esquecer sua senha.";
	mes "Aqui está:";
	mes "^0000FF"+#RecoveryPhrase$+"^000000";
	mes "ANOTE-A!!";
	set @AE_Seguro,1;
	close;

iSI:
	next;
	mes .@npcName$;
	mes "Esse é um Sistema único que permitirá a você ter uma segunda senha para sua conta.";
	mes "Em outras palavras, depois de logar, a ^FF0000Guardiã da Segurança^000000 pedirá que você digite sua Senha de Segurança.";
	mes "Isso irá prevenir que outros invadam sua conta enquanto você estiver fora.";
	next;
	mes .@npcName$;
	mes "Isto é tudo para nosso Sistema.";
	mes "Então, o que você gostaria de fazer?";
	callfunc "Set_AcctSecure";
	end;
	cancel:
	close;
}
