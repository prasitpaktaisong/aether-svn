function	script	InsZorya	{
		dispbottom "[Insígnia de Zorya]";
		set @inszorya, rand(10,50);
		set #AE_EvtPt, #AE_EvtPt + @inszorya;
		dispbottom "- Você adquiriu "+@inszorya+" Pontos de Evento."; 
		dispbottom "- Atualmente, você possui "+#AE_EvtPt+" Pontos de Evento.";
		end;
}
